            Dictionary<int, Gemeente> gemeenten = new Dictionary<int, Gemeente>();

            using GenericParser gemeentenReader = new GenericParser(Config.Path + "/" + Config.Gemeentenamen)
            {
                ColumnDelimiter = ';',
                FirstRowHasHeader = true,
                MaxBufferSize = 4096 // hiermee mag je spelen
            };
            while (gemeentenReader.Read())
            {
                int naamId = int.Parse(gemeentenReader[0].Trim());
                gemeenten.Add(naamId, 
                    new Gemeente 
                    { 
                        NaamId = naamId, 
                        Id = int.Parse(gemeentenReader[1].Trim()), 
                        TaalCode = gemeentenReader[2].Trim(), 
                        Naam = gemeentenReader[3].Trim() 
                    });                
            }
